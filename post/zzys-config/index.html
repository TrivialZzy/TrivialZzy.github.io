vim.keymap.set('', 'H', '^', {remap = true})
vim.keymap.set('', 'L', '$', {remap = true})

-- 选择上一个高亮并居中
vim.keymap.set('', 'N', 'Nzz')
-- 取消高亮
vim.keymap.set('', '<leader><cr>', '<cmd>nohlsearch<cr>', {remap = true})
-- 可视模式下复制到系统剪贴板
vim.keymap.set('v', 'y', '"+y')
-- 全选且复制
vim.keymap.set('n', 'Y', 'ggVG"+y')
-- 缩进设置
vim.keymap.set('n', '>', '>>', {remap = true})
vim.keymap.set('n', '<', '<<', {remap = true})
-- 保存
vim.keymap.set('', 'S', '<cmd>w<cr>')
-- 退出
vim.keymap.set('', 'Q', '<cmd>q<cr>')
-- 禁用 s 键
vim.keymap.set('', 's', '<Nop>', {remap = true})

local lazypath = vim.fn.stdpath("data") .. "/lazy/lazy.nvim"

    vim.fn.system({
        "git",
        "clone",
        "--filter=blob:none",
        "https://github.com/folke/lazy.nvim.git",
        "--branch=stable", -- latest stable release
        lazypath,
    })
end
vim.opt.rtp:prepend(lazypath)
require("lazy").setup("plugins")

return {
    "f-person/auto-dark-mode.nvim",
    config = {
        update_interval = 1000,
        set_dark_mode = function()
            vim.api.nvim_set_option("background", "dark")
            vim.cmd("colorscheme gruvbox")
        end,
        set_light_mode = function()
            vim.api.nvim_set_option("background", "light")
            vim.cmd("colorscheme gruvbox")
        end,
    },
    init = function()
        require("auto-dark-mode").init()
    end,
    cond = function()
        if vim.g.vscode then
            return false
        else
            return true
        end
    end
}

let mapleader=' ' " 设置 Leader 键
" set clipboard=unnamed " 系统剪贴板

" 移动设置
noremap J 5j
noremap K 5k
noremap H ^
noremap L $
noremap s <nop>
" 取消高亮
noremap <LEADER><CR> :nohlsearch<CR>
" 缩进设置
nnoremap > >>
nnoremap < <<
" 可视模式下系统剪贴板设置
vnoremap y "+y
" 全选复制
noremap Y ggVG"+y

map S :w<CR>
map Q :q<CR>

set lazyredraw

" 选择下一个高亮并居中
noremap n nzz
" 选择上一个高亮并居中
noremap N Nzz


if exists('g:vscode')
    " VSCode extension

    " 行注释
    vnoremap <C-/> <cmd>call VSCodeNotifyVisual('noop', 1)<CR><Esc><cmd>call VSCodeNotify('editor.action.commentLine')<CR>
    " 使用 VSCode 的搜索
    vnoremap <C-f> <cmd>call VSCodeNotifyVisual('noop', 1)<CR><Esc><cmd>call VSCodeNotify('actions.find')<CR>

    " 插件
    call plug#begin('~/.config/nvim/plugged')
 
    Plug 'tpope/vim-surround'
    Plug 'ybian/smartim'
 
    call plug#end()

else
    " ordinary Neovim

    syntax on " 语法高亮
    set scrolloff=5 " 内容上下保存 5 个单位
    set number " 行号
    set relativenumber " 相对行号
    " 缩进设置 begin
    set tabstop=4
    set softtabstop=4
    set shiftwidth=4
    set expandtab
    " 缩进设置 end
    set wildmenu " 更好的命令行补全
    set cursorline " 高亮行
    set wrap " 一行内容不超出屏幕

    map R :so %<CR>

    " 插件
    call plug#begin('~/.config/nvim/plugged')

    Plug 'jiangmiao/auto-pairs'
    Plug 'tpope/vim-surround'
    Plug 'arzg/vim-colors-xcode'
    Plug 'ybian/smartim'

    call plug#end()

    " 主题
    colorscheme xcode

endif
