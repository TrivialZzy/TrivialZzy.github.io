class Solution {
public:
    string baseNeg2(int n) {
        if (n == 0) {
            return "0";
        }

        int mod = -2; // k进制替换即可
        string ans;
        
        while (n) {
            int t = n%mod;
            n /= mod;
            if (t < 0) {
                t -= mod;
                n += 1;
            }
            ans.push_back('0'+t);
        }
        reverse(ans.begin(), ans.end());

        return ans;
    }
};

注解

注意任意进制的转换.

class Solution {
public:
    int storeWater(vector<int>& bucket, vector<int>& vat) {
        int M = *max_element(vat.begin(), vat.end());
        int ans = INT_MAX;

        if (M == 0) {
            return 0;
        }

        for (int i=1; i<=M; i++) {
            int t = 0;
            for (int j=0; j<bucket.size(); j++) {
                t += max(0, (vat[j]+i-1)/i-bucket[j]);
            }
            ans = min(ans, t+i);
        }

        return ans;
    }
};

值得注意的是上取整: 对于 $x, y\in\mathbb{Z}_+$, $\lceil y/x \rceil = int((y+x-1)/x)$.

class Solution {
public:
    int mctFromLeafValues(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));

        for (int i=n-1; i>=0; i--){
            dp[i][i] = 0;
            for (int j=i+1; j<n; j++) {
                for (int k=i; k<j; k++) {
                    int M1 = *max_element(arr.begin()+i, arr.begin()+k+1);
                    int M2 = *max_element(arr.begin()+k+1, arr.begin()+j+1);
                    dp[i][j] = min(dp[i][j], dp[i][k] + dp[k+1][j] + M1*M2);
                }
            }
        }

        return dp[0][n-1];
    }
};

注解

注意区间DP时的枚举顺序, 是正序还是倒序.

class Solution {
public:
    int maximumTastiness(vector<int>& price, int k) {
        sort(price.begin(), price.end());

        auto f = [&](int d) {
            int cnt = 1, t = price[0];
            for (auto it : price) {
                if (it >= t+d) {
                    cnt++;
                    t = it;
                }
            }
            return cnt;
        };

        int l = 0, r = (price[price.size()-1]-price[0])/(k-1) + 1;
        while (l+1 < r) {
            int mid = (l+r) >> 1;
            if (f(mid) >= k) {
                l = mid;
            } else {
                r = mid;
            }
        }

        return l;
    }
};

注解

任意两者绝对值最小(大) == 排序后相邻两者绝对值最小(大).

注意套路: 最大化最小值/最小化最大值 -> 二分答案.

class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        ans, s = 0, 0
        for t in nums:
            if s == 0:
                ans = t
                s = 1
            else:
                s += (1 if t==ans else -1)
        return ans

注解

在O(1)空间复杂度的情况下求出众数(要求众数的数量大于数据的一半).
